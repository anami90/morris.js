Custom column chart

var data = {
  '1M': 0.2504,
  '3M': -0.6916,
  '6M': 0.7020,
  '1Y': -0.0445,
  '3Y': 0.5849,
  '5Y': 0.7928
};

var chart = new Extra.Column({
  data: data,
  element: '#graph',
  precision: 0
});

var inversed = function(){
  chart.update({
    '1M': -0.2504,
  '3M': 0.6916,
  '6M': -0.7020,
  '1Y': 0.0445,
  '3Y': -0.5849,
  '5Y': -0.8228
  });
};

var regular = function(){
  chart.update(data);
};

setInterval(inversed, 4000);
setTimeout(function(){ setInterval(regular, 4000); }, 2000);
